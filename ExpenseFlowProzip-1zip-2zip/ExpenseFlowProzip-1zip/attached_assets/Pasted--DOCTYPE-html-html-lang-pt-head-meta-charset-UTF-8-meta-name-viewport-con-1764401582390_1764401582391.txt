    <script type="module">
        import { initializeApp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js';
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js';
        import { getFirestore, doc, setDoc, getDoc, updateDoc, onSnapshot, arrayUnion, collection } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js';

        // --- CHECKLIST #1: Valida√ß√£o de Ambiente ---
        function checkEnvironment() {
            if (!window.__firebase_config) {
                document.body.innerHTML = `
                    <div style="height:100vh; display:flex; flex-direction:column; align-items:center; justify-content:center; background:#1a1a2e; color:white; text-align:center; padding:20px; font-family:sans-serif;">
                        <div style="font-size:3rem; margin-bottom:20px;">‚ö†Ô∏è</div>
                        <h1 style="color:#ef4444; font-size:2rem; margin-bottom:10px;">Configura√ß√£o Ausente</h1>
                        <p>Para jogar, voc√™ precisa conectar ao Firebase.</p>
                        <div style="background:#000; padding:15px; border-radius:8px; text-align:left; margin-top:20px; font-family:monospace; color:#4ade80; max-width:500px; font-size:0.85rem;">
                            1. Abra este arquivo num editor de texto.<br>
                            2. Procure por "√ÅREA DE CONFIGURA√á√ÉO" no topo.<br>
                            3. Descomente as linhas e cole suas chaves do Firebase.<br>
                            4. Salve e recarregue a p√°gina.
                        </div>
                    </div>
                `;
                throw new Error("Firebase Config missing");
            }
        }

        // Configura√ß√£o Global e Vari√°veis de Estado
        window.gameState = {
            user: null,
            roomCode: '',
            isHost: false,
            unsubscribe: null, // Para parar de ouvir o banco quando sair
            currentDocData: null
        };

        // Inicializa√ß√£o Segura
        checkEnvironment();
        
        const firebaseConfig = JSON.parse(window.__firebase_config);
        const appId = typeof window.__app_id !== 'undefined' ? window.__app_id : 'default-app-id';
        
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);

        // Banco de Palavras
        const wordBank = {
            "Lugar": ["Praia", "Escola", "Hospital", "Cemit√©rio", "Shopping", "Cinema", "Zool√≥gico", "Igreja", "Teatro", "Aeroporto", "Academia", "Padaria"],
            "Objeto": ["Televis√£o", "Garfo", "Smartphone", "Guarda-chuva", "Martelo", "Viol√£o", "Cadeira", "Rel√≥gio", "Livro", "Espelho", "Caneta", "Sapato"],
            "Comida": ["Pizza", "Sushi", "Hamburguer", "Sorvete", "Lasanha", "Salada", "Ovo Frito", "Chocolate", "Pipoca", "Feijoada", "Macarr√£o", "Bolo"]
        };

        // --- AUTENTICA√á√ÉO ---
        async function initAuth() {
            try {
                // Checklist #2: Prioridade para Token Customizado (Ambiente Seguro)
                if (typeof window.__initial_auth_token !== 'undefined' && window.__initial_auth_token) {
                    await signInWithCustomToken(auth, window.__initial_auth_token);
                    console.log("Autenticado com token seguro.");
                } else {
                    await signInAnonymously(auth);
                    console.log("Autenticado anonimamente.");
                }
            } catch (error) {
                console.error("Erro Cr√≠tico Auth:", error);
                showToast("Erro de conex√£o com o servidor.");
            }
        }

        onAuthStateChanged(auth, (user) => {
            if (user) {
                window.gameState.user = user;
                // Checklist #6: Garantir que o DOM est√° pronto antes de interagir
                if (document.readyState === 'loading') {
                    document.addEventListener('DOMContentLoaded', () => handleUserReady(user));
                } else {
                    handleUserReady(user);
                }
            }
        });

        function handleUserReady(user) {
            console.log("Pronto para jogar. ID:", user.uid);
            checkUrlHash(); 
            setLoading(false);
        }

        // --- FIREBASE FIRESTORE ---

        // Caminho Rigoroso: /artifacts/{appId}/public/data/rooms/{code}
        const getRoomRef = (code) => doc(db, 'artifacts', appId, 'public', 'data', 'rooms', code);

        window.criarSala = async function() {
            if (!validarNome()) return;
            if (!window.gameState.user) { showToast("Conectando..."); return; }
            
            setLoading(true);
            const code = Math.random().toString(36).substring(2, 6).toUpperCase();
            const userId = window.gameState.user.uid;
            const username = document.getElementById('username').value.trim();

            try {
                await setDoc(getRoomRef(code), {
                    hostId: userId,
                    status: 'waiting',
                    players: [{ uid: userId, name: username }],
                    createdAt: Date.now(),
                    currentWord: '',
                    currentCategory: '',
                    impostorId: ''
                });

                window.gameState.isHost = true;
                entrarNaLogicaDaSala(code, username);
            } catch (error) {
                console.error("Erro Criar Sala:", error);
                if (error.code === 'permission-denied') {
                    showToast("Erro de permiss√£o. Recarregue a p√°gina.");
                } else {
                    showToast("Erro ao criar sala.");
                }
            } finally {
                setLoading(false);
            }
        };

        window.entrarSala = async function() {
            if (!validarNome()) return;
            if (!window.gameState.user) { showToast("Conectando..."); return; }
            
            const codeInput = document.getElementById('roomCodeInput');
            const code = codeInput.value.trim().toUpperCase();
            
            if (code.length < 4) { showToast("C√≥digo inv√°lido."); return; }

            setLoading(true);
            const userId = window.gameState.user.uid;
            const username = document.getElementById('username').value.trim();

            try {
                const roomRef = getRoomRef(code);
                const roomSnap = await getDoc(roomRef);

                if (!roomSnap.exists()) {
                    showToast("Sala n√£o encontrada!");
                    setLoading(false);
                    return;
                }

                // Evita duplicatas no array
                const currentPlayers = roomSnap.data().players || [];
                const isAlreadyIn = currentPlayers.some(p => p.uid === userId);

                if (!isAlreadyIn) {
                    await updateDoc(roomRef, {
                        players: arrayUnion({ uid: userId, name: username })
                    });
                }

                // Verifica se o user atual √© o host (caso tenha recarregado a p√°gina)
                window.gameState.isHost = (roomSnap.data().hostId === userId);
                entrarNaLogicaDaSala(code, username);

            } catch (error) {
                console.error("Erro Entrar Sala:", error);
                showToast("Erro ao entrar. Verifique o c√≥digo.");
            } finally {
                setLoading(false);
            }
        };

        function entrarNaLogicaDaSala(code, username) {
            window.gameState.roomCode = code;
            window.location.hash = code;

            if (window.gameState.unsubscribe) window.gameState.unsubscribe();

            window.gameState.unsubscribe = onSnapshot(getRoomRef(code), (doc) => {
                if (doc.exists()) {
                    renderizarEstadoDaSala(doc.data());
                } else {
                    sairDoJogo(true); 
                    showToast("A sala foi encerrada.");
                }
            }, (error) => {
                console.error("Erro Listener:", error);
            });

            showScreen('screen-lobby');
        }

        // --- L√ìGICA DE JOGO (HOST) ---

        window.iniciarJogo = async function() {
            if (!window.gameState.user) return;
            const code = window.gameState.roomCode;
            
            if (!window.gameState.currentDocData) return;
            const players = window.gameState.currentDocData.players;

            if (players.length < 3) {
                showToast("M√≠nimo de 3 jogadores!");
                return;
            }

            // Sorteio
            const categories = Object.keys(wordBank);
            const randomCat = categories[Math.floor(Math.random() * categories.length)];
            const words = wordBank[randomCat];
            const randomWord = words[Math.floor(Math.random() * words.length)];
            
            const impostorIndex = Math.floor(Math.random() * players.length);
            const impostorId = players[impostorIndex].uid;

            try {
                await updateDoc(getRoomRef(code), {
                    status: 'playing',
                    currentCategory: randomCat,
                    currentWord: randomWord,
                    impostorId: impostorId
                });
            } catch (e) {
                showToast("Erro ao iniciar partida.");
            }
        };

        window.voltarLobbyHost = async function() {
            if (!window.gameState.user) return;
            try {
                await updateDoc(getRoomRef(window.gameState.roomCode), {
                    status: 'waiting',
                    currentWord: '',
                    impostorId: ''
                });
            } catch (e) { console.error(e); }
        };

        // --- UI REATIVA ---

        function renderizarEstadoDaSala(data) {
            window.gameState.currentDocData = data;
            const myId = window.gameState.user.uid;

            // Renderizar Lista
            const list = document.getElementById('playersList');
            if (list) {
                list.innerHTML = '';
                document.getElementById('displayRoomCode').innerText = window.gameState.roomCode;

                data.players.forEach(p => {
                    const li = document.createElement('li');
                    li.className = "bg-gray-800 p-3 rounded-lg flex items-center justify-between border border-gray-700 transform transition-all duration-300 hover:scale-[1.02]";
                    const initial = p.name ? p.name.charAt(0).toUpperCase() : '?';
                    const isMe = p.uid === myId;
                    const isHost = p.uid === data.hostId;
                    
                    li.innerHTML = `
                        <div class="flex items-center gap-3">
                            <div class="w-8 h-8 rounded-full ${isMe ? 'bg-blue-600' : 'bg-gray-600'} flex items-center justify-center text-xs font-bold text-white shadow-inner ring-2 ${isMe ? 'ring-blue-400' : 'ring-transparent'}">
                                ${initial}
                            </div>
                            <div class="flex flex-col">
                                <span class="text-white font-medium ${isMe ? 'text-blue-300' : ''}">${p.name} ${isMe ? '(Voc√™)' : ''}</span>
                                ${isHost ? '<span class="text-[10px] text-yellow-500 uppercase tracking-wider font-bold"><i class="fa-solid fa-crown"></i> Host</span>' : ''}
                            </div>
                        </div>
                    `;
                    list.appendChild(li);
                });
            }

            // Gerenciar Telas
            if (data.status === 'playing') {
                if (!document.getElementById('screen-game').classList.contains('active')) {
                    setupGameScreen(data, myId);
                }
            } else {
                if (!document.getElementById('screen-lobby').classList.contains('active')) {
                    showScreen('screen-lobby');
                }
                
                const amIHost = data.hostId === myId;
                window.gameState.isHost = amIHost;

                if (amIHost) {
                    document.getElementById('hostControls').classList.remove('hidden');
                    document.getElementById('waitingMessage').classList.add('hidden');
                } else {
                    document.getElementById('hostControls').classList.add('hidden');
                    document.getElementById('waitingMessage').classList.remove('hidden');
                }
            }
        }

        function setupGameScreen(data, myId) {
            const isImpostor = (myId === data.impostorId);
            const card = document.getElementById('gameCard');
            const innocentView = document.getElementById('innocentView');
            const impostorView = document.getElementById('impostorView');
            const hostControls = document.getElementById('hostGameControls');

            resetVision();

            // Reset Classes
            card.classList.remove('impostor-theme', 'innocent-theme');
            innocentView.classList.add('hidden');
            impostorView.classList.add('hidden');
            innocentView.classList.remove('flex');
            impostorView.classList.remove('flex');

            if (isImpostor) {
                card.classList.add('impostor-theme');
                impostorView.classList.remove('hidden');
                impostorView.classList.add('flex');
            } else {
                card.classList.add('innocent-theme');
                document.getElementById('categoryDisplay').innerText = data.currentCategory;
                document.getElementById('secretWordDisplay').innerText = data.currentWord;
                innocentView.classList.remove('hidden');
                innocentView.classList.add('flex');
            }

            if (window.gameState.user && data.hostId === window.gameState.user.uid) {
                hostControls.classList.remove('hidden');
            } else {
                hostControls.classList.add('hidden');
            }

            showScreen('screen-game');
        }

        // --- INICIALIZA√á√ÉO ---
        initAuth();

        // --- HELPERS GLOBAIS ---
        window.copiarLink = function() {
            const url = window.location.href.split('#')[0] + '#' + window.gameState.roomCode;
            navigator.clipboard.writeText(url).then(() => showToast("Link copiado!"));
        }

        function checkUrlHash() {
            const hash = window.location.hash.substring(1);
            if (hash && hash.length >= 4) {
                const codeInput = document.getElementById('roomCodeInput');
                if(codeInput) codeInput.value = hash;
            }
        }

        // --- JAVASCRIPT UI ---

        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(s => {
                s.classList.remove('active');
                setTimeout(() => { if(!s.classList.contains('active')) s.style.display = 'none'; }, 300);
            });
            const screen = document.getElementById(screenId);
            screen.style.display = 'flex';
            setTimeout(() => screen.classList.add('active'), 50);
        }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMsg').innerText = msg;
            toast.style.opacity = '1';
            toast.style.transform = 'translate(-50%, 0)';
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.transform = 'translate(-50%, -20px)';
            }, 3000);
        }

        function setLoading(isLoading) {
            const overlay = document.getElementById('loadingOverlay');
            overlay.style.opacity = isLoading ? '1' : '0';
            overlay.style.pointerEvents = isLoading ? 'all' : 'none';
        }

        function validarNome() {
            const nomeInput = document.getElementById('username');
            if (!nomeInput.value.trim()) {
                showToast("Como devemos te chamar?");
                nomeInput.focus();
                nomeInput.classList.add('ring-2', 'ring-red-500');
                setTimeout(() => nomeInput.classList.remove('ring-2', 'ring-red-500'), 2000);

            visionHidden = false;
